import urllib.request
import json
import time

def check_grid_and_run():
    # VoltWise API 서버 호출
    url = "http://localhost:8000/api/status/live"
    try:
        req = urllib.request.urlopen(url)
        res = json.loads(req.read().decode())
        if res.get("success") and res["rating"]["code"] in ["GREEN", "YELLOW"]:
            print("🔋 ESG 전력망 수급 요건 충족. 학습 작업을 시작합니다.")
            return True
        else:
            print("🚫 전력수급 오렌지/레드 주의 상태. 탄소 저감을 위해 30분 연기합니다.")
            return False
    except Exception as e:
        print(f"Fallback 시작: {e}")
        return True # API 장애 시에는 기본 실행 허용

if __name__ == '__main__':
    while not check_grid_and_run():
        time.sleep(1800) # 30분 대기 후 재호출
    # python train.py 시작 로직 작성

name: ESG ML Training Scheduler
on:
  push:
    branches: [ main ]

jobs:
  train:
    runs-on: ubuntu-latest
    steps:
      - name: Checkout Code
        uses: actions/checkout@v3

      - name: Check VoltWise Live Carbon Status
        run: |
          # VoltWise 클라우드 배포 서버로 수급 상황 검증
          STATUS=$(curl -s "http://localhost:8000/api/status/live" | jq -r .rating.code)
          echo "Current Grid Status is $STATUS"
          if [ "$STATUS" == "RED" ] || [ "$STATUS" == "ORANGE" ]; then
            echo "Grid stress is high. Delaying runner workflow."
            exit 1 # 파이프라인 일시 차단
          fi
          
      - name: Execute ML Training
        run: python src/train.py
