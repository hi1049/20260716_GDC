# 실시간 전력 상태의 탄소 인식(Carbon-Aware) 추천이 RUN_NOW(즉시 기동)일 경우에만 AI 컨테이너 가동
RECOMMENDATION=$(curl -s http://localhost:8000/api/advice | jq -r .recommendation)

if [ "$RECOMMENDATION" = "RUN_NOW" ]; then
  echo "✅ 전력망 공급 충분 및 탄소 밀도 최저. 대규모 LLM 파인튜닝 컨테이너를 가동합니다."
  docker run --gpus all -d llm-training-job:latest
else
  DEFER_TIME=$(curl -s http://localhost:8000/api/advice | jq -r .defer_until_hint)
  echo "⚠️ 현재 고탄소 발전 시간대. 작업을 보류하며, [$DEFER_TIME] 가동을 권장합니다."
  echo "작업 가동을 지연 대기 큐로 이전합니다."
fi
import urllib.request
import json
import sys

def check_carbon_grid():
    try:
        req = urllib.request.Request("http://localhost:8000/api/advice", headers={'User-Agent': 'GridPulse-Agent'})
        with urllib.request.urlopen(req, timeout=5) as response:
            res = json.loads(response.read().decode('utf-8'))
            
            if res.get("recommendation") == "RUN_NOW":
                print(f"✅ 탄소 인식 가동 승인: {res.get('reasoning')}")
                print(f"절감 효과 예측: {res.get('estimated_saving')}")
                sys.exit(0) # 가동 수행 승인 코드
            else:
                print(f"❌ 가동 보류 권고: {res.get('reasoning')}")
                print(f"개시 권장 시점: {res.get('defer_until_hint')}")
                sys.exit(1) # 지연 및 대기 코드
    except Exception as e:
        print(f"⚠️ API 장애 발생, 무중단 기본 실행 모드 작동: {e}")
        sys.exit(0)

if __name__ == "__main__":
    check_carbon_grid()
name: Carbon-Aware GPU Training
on:
  workflow_dispatch:

jobs:
  train:
    runs-on: self-hosted-gpu-runner
    steps:
      - name: Check Carbon Grid Advisor
        id: check_grid
        run: |
          ADVICE=$(curl -s http://localhost:8000/api/advice)
          DECISION=$(echo $ADVICE | jq -r .recommendation)
          echo "decision=$DECISION" >> $GITHUB_OUTPUT
          
      - name: Defer Training If Grid Heavy
        if: steps.check_grid.outputs.decision != 'RUN_NOW'
        run: |
          echo "그리드가 혼잡하거나 발전원별 탄소 강도가 높아 작업을 슬립 상태로 지연시킵니다."
          exit 1
          
      - name: Execute High Load GPU Train Job
        run: |
          python train_model.py --epochs 100 --batch-size 128
